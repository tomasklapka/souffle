39  void setup() override {
40  // for each clause, store a mapping from the head relation name to body relation names
41  for (auto& rel : prog.getAllRelations()) {
42  std::string name = rel->getName();
43 
44  // only process info relations
45  if (name.find("@info") == std::string::npos) {
46  continue;
47  }
48 
49  // find all the info tuples
50  for (auto& tuple : *rel) {
51  std::vector<std::string> bodyRels;
52 
53  RamDomain ruleNum;
54  tuple >> ruleNum;
55 
56  for (size_t i = 1; i < rel->getArity() - 1; i++) {
57  std::string bodyRel;
58  tuple >> bodyRel;
59  bodyRels.push_back(bodyRel);
60  }
61 
62  std::string rule;
63  tuple >> rule;
64 
65  info.insert({std::make_pair(name.substr(0, name.find(".@info")), ruleNum), bodyRels});
66  rules.insert({std::make_pair(name.substr(0, name.find(".@info")), ruleNum), rule});
67  }
68  }

71  std::unique_ptr<TreeNode> explain(
72  std::string relName, std::vector<RamDomain> tuple, int ruleNum, int levelNum, size_t depthLimit) {
73  std::stringstream joinedArgs;
74  joinedArgs << join(numsToArgs(relName, tuple), ", ");
75  auto joinedArgsStr = joinedArgs.str();
76 
77  // if fact
78  if (levelNum == 0) {
79  return std::make_unique<LeafNode>(relName + "(" + joinedArgsStr + ")");
80  }
81 
82  assert(info.find(std::make_pair(relName, ruleNum)) != info.end() && "invalid rule for tuple");
83 
84  // if depth limit exceeded
85  if (depthLimit <= 1) {
86  tuple.push_back(ruleNum);
87  tuple.push_back(levelNum);
88 
89  // find if subproof exists already
90  size_t idx = 0;
91  auto it = std::find(subproofs.begin(), subproofs.end(), tuple);
92  if (it != subproofs.end()) {
93  idx = it - subproofs.begin();
94  } else {
95  subproofs.push_back(tuple);
96  idx = subproofs.size() - 1;
97  }
98 
99  return std::make_unique<LeafNode>("subproof " + relName + "(" + std::to_string(idx) + ")");
100  }
101 
102  auto internalNode = std::make_unique<InnerNode>(
103  relName + "(" + joinedArgsStr + ")", "(R" + std::to_string(ruleNum) + ")");
104 
105  // make return vector pointer
106  std::vector<RamDomain> ret;
107  std::vector<bool> err;
108 
109  // add level number to tuple
110  tuple.push_back(levelNum);
111 
112  // execute subroutine to get subproofs
113  prog.executeSubroutine(relName + "_" + std::to_string(ruleNum) + "_subproof", tuple, ret, err);
114 
115  // recursively get nodes for subproofs
116  size_t tupleCurInd = 0;
117  for (std::string bodyRel : info[std::make_pair(relName, ruleNum)]) {
118  // check whether the current atom is a constraint
119  bool isConstraint =
120  std::find(constraintList.begin(), constraintList.end(), bodyRel) != constraintList.end();
121 
122  // handle negated atom names
123  auto bodyRelAtomName = bodyRel;
124  if (bodyRel[0] == '!') {
125  bodyRelAtomName = bodyRel.substr(1);
126  }
127 
128  // traverse subroutine return
129  size_t arity;
130  if (isConstraint) {
131  // we only handle binary constraints, and assume arity is 4 to account for hidden provenance
132  // annotations
133  arity = 4;
134  } else {
135  arity = prog.getRelation(bodyRelAtomName)->getArity();
136  }
137  auto tupleEnd = tupleCurInd + arity;
138 
139  // store current tuple and error
140  std::vector<RamDomain> subproofTuple;
141  std::vector<bool> subproofTupleError;
142 
143  for (; tupleCurInd < tupleEnd - 2; tupleCurInd++) {
144  subproofTuple.push_back(ret[tupleCurInd]);
145  subproofTupleError.push_back(err[tupleCurInd]);
146  }
147 
148  int subproofRuleNum = ret[tupleCurInd];
149  int subproofLevelNum = ret[tupleCurInd + 1];
150 
151  // for a negation, display the corresponding tuple and do not recurse
152  if (bodyRel[0] == '!') {
153  std::stringstream joinedTuple;
154  joinedTuple << join(numsToArgs(bodyRelAtomName, subproofTuple, &subproofTupleError), ", ");
155  auto joinedTupleStr = joinedTuple.str();
156  internalNode->add_child(std::make_unique<LeafNode>(bodyRel + "(" + joinedTupleStr + ")"));
157  internalNode->setSize(internalNode->getSize() + 1);
158  // for a binary constraint, display the corresponding values and do not recurse
159  } else if (isConstraint) {
160  std::stringstream joinedConstraint;
161 
162  if (isNumericBinaryConstraintOp(toBinaryConstraintOp(bodyRel))) {
163  joinedConstraint << subproofTuple[0] << " " << bodyRel << " " << subproofTuple[1];
164  } else {
165  joinedConstraint << bodyRel << "(\"" << prog.getSymbolTable().resolve(subproofTuple[0])
166  << "\", \"" << prog.getSymbolTable().resolve(subproofTuple[1]) << "\")";
167  }
168 
169  internalNode->add_child(std::make_unique<LeafNode>(joinedConstraint.str()));
170  internalNode->setSize(internalNode->getSize() + 1);
171  // otherwise, for a normal tuple, recurse
172  } else {
173  auto child =
174  explain(bodyRel, subproofTuple, subproofRuleNum, subproofLevelNum, depthLimit - 1);
175  internalNode->setSize(internalNode->getSize() + child->getSize());
176  internalNode->add_child(std::move(child));
177  }
178 
179  tupleCurInd = tupleEnd;
180  }
181 
182  return std::move(internalNode);
183  }

185  std::unique_ptr<TreeNode> explain(
186  std::string relName, std::vector<std::string> args, size_t depthLimit) override {
187  auto tuple = argsToNums(relName, args);
188  if (tuple.empty()) {
189  return std::make_unique<LeafNode>("Relation not found");
190  }
191 
192  std::pair<int, int> tupleInfo = findTuple(relName, tuple);
193  int ruleNum = tupleInfo.first;
194  int levelNum = tupleInfo.second;
195 
196  if (ruleNum < 0 || levelNum == -1) {
197  return std::make_unique<LeafNode>("Tuple not found");
198  }
199 
200  return explain(relName, tuple, ruleNum, levelNum, depthLimit);
