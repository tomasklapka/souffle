35 std::vector<std::unique_ptr<RamCondition>> getConditions(const RamCondition* condition) {
36  std::vector<std::unique_ptr<RamCondition>> conditions;
37  while (condition != nullptr) {
38  if (const auto* ramConj = dynamic_cast<const RamConjunction*>(condition)) {
39  conditions.emplace_back(ramConj->getRHS().clone());
40  condition = &ramConj->getLHS();

179 std::unique_ptr<RamExpression> CreateIndicesTransformer::getIndexElement(
180  RamCondition* c, size_t& element, size_t identifier) {
181  if (auto* binRelOp = dynamic_cast<RamConstraint*>(c)) {
182  if (binRelOp->getOperator() == BinaryConstraintOp::EQ) {
183  if (auto* lhs = dynamic_cast<RamElementAccess*>(binRelOp->getLHS())) {
184  RamExpression* rhs = binRelOp->getRHS();
185  if (rvla->getLevel(lhs) == identifier &&
186  (rcva->isConstant(rhs) || rvla->getLevel(rhs) < identifier)) {
187  element = lhs->getElement();
188  return std::unique_ptr<RamExpression>(rhs->clone());
189  }
190  }
191  if (auto* rhs = dynamic_cast<RamElementAccess*>(binRelOp->getRHS())) {
192  RamExpression* lhs = binRelOp->getLHS();
193  if (rvla->getLevel(rhs) == identifier &&
194  (rcva->isConstant(lhs) || rvla->getLevel(lhs) < identifier)) {
195  element = rhs->getElement();
196  return std::unique_ptr<RamExpression>(lhs->clone());
197  }
198  }
199  }
200  }
201  return nullptr;

204 std::unique_ptr<RamOperation> CreateIndicesTransformer::rewriteScan(const RamScan* scan) {
205  if (const auto* filter = dynamic_cast<const RamFilter*>(&scan->getOperation())) {
206  const RamRelation& rel = scan->getRelation();
207  const size_t identifier = scan->getIdentifier();
208 
209  // Values of index per column of table (if indexable)
210  std::vector<std::unique_ptr<RamExpression>> queryPattern(rel.getArity());
211 
212  // Remaining conditions which weren't handled by an index
213  std::unique_ptr<RamCondition> condition;
214 
215  auto addCondition = [&](std::unique_ptr<RamCondition> c) {
216  if (condition != nullptr) {
217  condition = std::make_unique<RamConjunction>(std::move(condition), std::move(c));
218  } else {
219  condition = std::move(c);
220  }
221  };
222 
223  bool indexable = false;
224 
225  for (auto& cond : getConditions(&filter->getCondition())) {
226  size_t element = 0;
227  if (std::unique_ptr<RamExpression> value = getIndexElement(cond.get(), element, identifier)) {
228  indexable = true;
229  if (queryPattern[element] == nullptr) {
230  queryPattern[element] = std::move(value);
231  } else {
232  addCondition(std::move(cond));
233  }
234  } else {
235  addCondition(std::move(cond));
236  }
237  }
238 
239  if (indexable) {
240  // replace scan by index scan
241  return std::make_unique<RamIndexScan>(std::make_unique<RamRelationReference>(&rel), identifier,
242  std::move(queryPattern),
243  condition == nullptr
244  ? std::unique_ptr<RamOperation>(filter->getOperation().clone())
245  : std::make_unique<RamFilter>(std::move(condition),
246  std::unique_ptr<RamOperation>(filter->getOperation().clone())),
247  scan->getProfileText());
248  }
249  }
250 
251  return nullptr;
252 }

330  bool dependsOn(const RamExpression* value, const size_t identifier) const {
331  std::vector<const RamExpression*> queue = {value};
332  while (!queue.empty()) {
333  const RamExpression* val = queue.back();
334  queue.pop_back();
335  if (const auto* elemAccess = dynamic_cast<const RamElementAccess*>(val)) {
336  if (context->rvla->getLevel(elemAccess) == identifier) {
337  return true;
338  }
339  } else if (const auto* intrinsicOp = dynamic_cast<const RamIntrinsicOperator*>(val)) {
340  for (const RamExpression* arg : intrinsicOp->getArguments()) {
341  queue.push_back(arg);
342  }
343  } else if (const auto* userDefinedOp = dynamic_cast<const RamUserDefinedOperator*>(val)) {
344  for (const RamExpression* arg : userDefinedOp->getArguments()) {
345  queue.push_back(arg);
346  }
347  }
348  }
349  return false;

359  std::unique_ptr<RamNode> operator()(std::unique_ptr<RamNode> node) const override {
360  if (auto* scan = dynamic_cast<RamRelationSearch*>(node.get())) {
361  const size_t identifier = scan->getIdentifier();
362  bool isExistCheck = true;
363  visitDepthFirst(scan->getOperation(), [&](const RamProject& project) {
364  if (isExistCheck) {
365  std::vector<const RamExpression*> values;
366  // TODO: function to extend vectors
367  const std::vector<RamExpression*> initialVals = project.getValues();
368  values.insert(values.end(), initialVals.begin(), initialVals.end());
369 
370  while (!values.empty()) {
371  const RamExpression* value = values.back();
372  values.pop_back();
373 
374  if (const auto* pack = dynamic_cast<const RamPackRecord*>(value)) {
375  const std::vector<RamExpression*> args = pack->getArguments();
376  values.insert(values.end(), args.begin(), args.end());
377  } else if (const auto* intrinsicOp =
378  dynamic_cast<const RamIntrinsicOperator*>(value)) {
379  for (auto* arg : intrinsicOp->getArguments()) {
380  values.push_back(arg);
381  }
382  } else if (value != nullptr && !context->rcva->isConstant(value) &&
383  context->rvla->getLevel(value) == identifier) {
384  isExistCheck = false;
385  break;
386  }
387  }
388  }
389  });
390  if (isExistCheck) {
391  visitDepthFirst(scan->getOperation(), [&](const RamUnpackRecord& lookup) {
392  if (isExistCheck) {
393  if (lookup.getReferenceLevel() == identifier) {
394  isExistCheck = false;
395  }
396  }
397  });
398  }
399  if (isExistCheck) {
400  visitDepthFirst(scan->getOperation(), [&](const RamExpression& expression) {
401  if (isExistCheck) {
402  if (dependsOn(&expression, identifier)) {
403  isExistCheck = false;
404  }
405  }
406  });
407  }
408  if (isExistCheck) {
409  // create constraint
410  std::unique_ptr<RamCondition> constraint;
411 
412  if (nullptr != dynamic_cast<RamScan*>(scan)) {
413  constraint = std::make_unique<RamNegation>(std::make_unique<RamEmptinessCheck>(
414  std::make_unique<RamRelationReference>(&scan->getRelation())));
415  } else if (auto* indexScan = dynamic_cast<RamIndexScan*>(scan)) {
416  std::vector<std::unique_ptr<RamExpression>> values;

421  values.push_back(nullptr);
422  }

425  std::make_unique<RamRelationReference>(&scan->getRelation()),
426  std::move(values));

430  std::unique_ptr<RamOperation>(scan->getOperation().clone()),
431  scan->getProfileText());
