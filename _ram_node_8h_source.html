146  * An abstract class for manipulating RAM Nodes by substitution
147  */
148 class RamNodeMapper {
149 public:
150  virtual ~RamNodeMapper() = default;
151 
152  /**
153  * Abstract replacement method for a node.
154  *
155  * If the given nodes is to be replaced, the handed in node
156  * will be destroyed by the mapper and the returned node
157  * will become owned by the caller.
158  */
159  virtual std::unique_ptr<RamNode> operator()(std::unique_ptr<RamNode> node) const = 0;
160 
161  /**
162  * Wrapper for any subclass of the RAM node hierarchy performing type casts.
163  */
164  template <typename T>
165  std::unique_ptr<T> operator()(std::unique_ptr<T> node) const {
166  std::unique_ptr<RamNode> resPtr =
167  (*this)(std::unique_ptr<RamNode>(static_cast<RamNode*>(node.release())));
168  assert(nullptr != dynamic_cast<T*>(resPtr.get()) && "Invalid target node!");
169  return std::unique_ptr<T>(dynamic_cast<T*>(resPtr.release()));
170  }
171 };
