34 /** get indexable element */
35 std::unique_ptr<RamExpression> RamAggregate::getIndexElement(
36  RamCondition* condition, size_t& element, size_t level) {
37  if (auto* binRelOp = dynamic_cast<RamConstraint*>(condition)) {
38  if (binRelOp->getOperator() == BinaryConstraintOp::EQ) {
39  if (auto* lhs = dynamic_cast<RamElementAccess*>(binRelOp->getLHS())) {
40  RamExpression* rhs = binRelOp->getRHS();
41  if (lhs->getIdentifier() == level &&
42  (RamConstValueAnalysis().isConstant(rhs) ||
43  RamExpressionLevelAnalysis().getLevel(rhs) < level)) {
44  element = lhs->getElement();
45  return std::unique_ptr<RamExpression>(rhs->clone());
46  }
47  }
48  if (auto* rhs = dynamic_cast<RamElementAccess*>(binRelOp->getRHS())) {
49  RamExpression* lhs = binRelOp->getLHS();
50  if (rhs->getIdentifier() == level &&
51  (RamConstValueAnalysis().isConstant(lhs) ||
52  RamExpressionLevelAnalysis().getLevel(lhs) < level)) {
53  element = rhs->getElement();
54  return std::unique_ptr<RamExpression>(lhs->clone());
55  }
56  }
57  }
58  }
59  return std::unique_ptr<RamExpression>(nullptr);
60 }
